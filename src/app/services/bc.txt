import { Injectable } from '@angular/core';
import { data } from './data.ts';
import { HttpClient } from '@angular/common/http';
import { Observable, of } from 'rxjs';

const configUrl = 'assets/data.json';

@Injectable({
  providedIn: 'root'
})
export class DatahandlerService {
  jsonData:Object = {};
  arrayData

  constructor(private http: HttpClient) {
    this.http.get(configUrl).subscribe((response)=>{
      this.jsonData = JSON.parse(JSON.stringify(response));
      console.log('service', this.jsonData);
    });
  }

  /**json data methods*/
  get jsonDataGetter():Object {
    return JSON.parse(JSON.stringify(this.jsonData));
  }

  getJsonData():Observable<Object> {
    return of(JSON.parse(JSON.stringify(this.jsonData)));
  }

  getObservable() {
    return this.http.get(configUrl);
  }
}
